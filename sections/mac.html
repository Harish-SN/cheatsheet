macOS

Handy macOS command-line utilities: Homebrew, system info, networking, processes, services, Finder, Spotlight, disk, and updates.

Homebrew (Package Manager) 
# Install Homebrew
/bin/bash -c "$(curl -fsSL https://raw.githubusercontent.com/Homebrew/install/HEAD/install.sh)"

# Update & upgrade
brew update
brew upgrade

# Install / remove package
brew install wget
brew uninstall wget

# Search package
brew search node

# List installed packages
brew list

# Services (start/stop daemons)
brew services start mysql
brew services stop mysql
brew services list
System Info 
# macOS version
sw_vers

# Hardware info
system_profiler SPHardwareDataType

# CPU & memory
sysctl -n machdep.cpu.brand_string
vm_stat

# Disk usage
df -h
diskutil list
Networking 
# Interfaces & IP addresses
ifconfig
ipconfig getifaddr en0

# Ping / Traceroute
ping -c 4 google.com
traceroute google.com

# DNS lookup
dig example.com
nslookup example.com

# Open a port scan (nc)
nc -zv 127.0.0.1 22-80
Processes & Monitoring 
# List processes
ps aux
top

# Kill process by PID
kill -9 1234

# Find process using a port
lsof -i :3000

# Activity Monitor (GUI)
open -a "Activity Monitor"
Launch Services (launchctl) 
# List all services
launchctl list

# Load / unload a plist
launchctl load ~/Library/LaunchAgents/myjob.plist
launchctl unload ~/Library/LaunchAgents/myjob.plist

# Start / stop service
launchctl start my.job
launchctl stop my.job
Finder & Files 
# Show hidden files in Finder
defaults write com.apple.finder AppleShowAllFiles -bool true && killall Finder

# Hide again
defaults write com.apple.finder AppleShowAllFiles -bool false && killall Finder

# Open folder/file in Finder
open .
open ~/Downloads

# Quick Look a file
qlmanage -p file.txt
Spotlight Search 
# Spotlight query
mdfind "hello world"

# Rebuild Spotlight index
sudo mdutil -E /

# Enable/disable Spotlight
sudo mdutil -i on /
sudo mdutil -i off /
Disk & Updates 
# Disk usage
df -h
du -sh *

# Check disk
diskutil verifyVolume /

# Repair disk
diskutil repairVolume /

# macOS software update
softwareupdate -l       # list
sudo softwareupdate -i -a   # install all
